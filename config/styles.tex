%шрифты
\defaultfontfeatures{Mapping=tex-text,Scale=MatchLowercase}
\setmainfont{Times New Roman}
\newfontfamily\cyrillicfont{Times New Roman}

%поля страницы
\usepackage[left=3cm,right=2cm,
top=2cm,bottom=2cm,bindingoffset=0cm]{geometry}
\sloppy

%текст
\parindent 1.25cm
\linespread{1.5}
\setlength\emergencystretch{3em}

% формат заголовочков
\titleformat{\section}[block]{\hspace{1.25cm}\bfseries}{\arabic{section}. }{0cm}{}
\titlespacing{\section}{0pt}{0pt}{0pt}
\titleformat{\subsection}[block]{\hspace{1.25cm}\bfseries}{\arabic{section}.\arabic{subsection}. }{0cm}{}
\titlespacing{\subsection}{0pt}{0pt}{0pt}

\titleformat{\subsubsection}[block]{\hspace{1.25cm}\bfseries}{\arabic{section}.\arabic{subsection}.\arabic{subsubsection}. }{0cm}{}
\titlespacing{\subsubsection}{0pt}{0pt}{0pt}

%формат списков
\setlist{nosep}
\setlist[enumerate]{wide=\parindent, label=\arabic*)}
\setlist[itemize]{wide=\parindent}
\renewcommand\labelitemi{--}

\renewcommand\UrlFont{\rmfamily} % стандартный шрифт для ссылок

% формат библиографии
\defbibenvironment{bibliography}
{\enumerate[label=\arabic*)]
	{}
	{\setlength{\leftmargin}{0pt}%
		\setlength{\itemindent}{1.25cm}%
		\setlength{\itemsep}{\bibitemsep}%
		\setlength{\parsep}{\bibparsep}}%
}
{\endenumerate}
{\item\addspace}

% формат подписей у изображений
\captionsetup[figure]{name={Рисунок },labelsep=endash}
% не добавлять отступ после подписи
\captionsetup{belowskip=-20pt}

\setmonofont{Courier New} % моноширинный шрифт

\fvset{listparameters=\setlength{\topsep}{5pt}\setlength{\partopsep}{0pt}}
